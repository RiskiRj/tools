## install on Termux

<pre>

$ pkg update && pkg upgrade
$ pkg install git
$ pkg install nodejs
$ npm init -y
$ npm install --save request colors readline shelljs
$ git clone https://github.com/ibnusyawall/mytools
$ cd mytools
$ node mytools

</pre>

## install on Linux / Ubuntu

<pre>

$ sudo apt-get install git
$ sudo apt-get install nodejs
$ npm init -y
$ npm install --save request colors readline shelljs
$ git clone https://github.com/ibnusyawall/mytools
$ cd mytools
$ node mytools

</pre>
